def self.use(type)
  database = CacheStoreDatabase.new(type)
  return_value = yield(database)
  database.close_if_open!
  return_value
end

def close_if_open!
  return unless @db

  cache_path.dirname.mkpath
  cache_path.atomic_write(JSON.dump(@db))
end
