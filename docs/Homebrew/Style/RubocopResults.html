def initialize(json)
  @metadata = json["metadata"]
  @file_offenses = {}
  json["files"].each do |f|
    next if f["offenses"].empty?

    file = File.realpath(f["path"])
    @file_offenses[file] = f["offenses"].map { |x| RubocopOffense.new(x) }
  end
end
